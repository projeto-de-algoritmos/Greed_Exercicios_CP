```cpp
void solve() {
    int n;
    string s;
    cin>>n>>s;
    int prev = s[0];
    int cnt = 1;
    int mx = 1;
    for(int i=1; i<n; i++) {
        if(s[i] == prev) {
            cnt++;
        } else {
            mx = max(cnt, mx);
            cnt = 1;
        }
        prev = s[i];
    }
    mx = max(cnt, mx);

    cout<<mx+1<<"\n";
}
```
